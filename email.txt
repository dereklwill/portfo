import smtplib
from email.message import EmailMessage
from string import Template
from pathlib import Path #os.path

name = 'Derek Williamson'
to = 'WilliamsonITBot@gmail.com'
html = Template(Path('email_template.txt').read_text())
email = EmailMessage()

email['from'] = name
email['to'] = to
email['subject'] = 'Test Python Email for contact form!'

email.set_content(html.substitute(name = 'BotTest'), 'html')

with smtplib.SMTP(host='smtp.gmail.com', port=587) as smtp:
    smtp.ehlo()
    smtp.starttls()
    smtp.login('WilliamsonITBot@gmail.com', 'Vivi1576!')
    smtp.send_message(email)
